```typescript
export interface PhoneTreeNode {
  id: string;
  prompt: string;                    // What the IVR says (for AI matching)
  action: "dtmf" | "speak" | "wait" | "skip";
  keys?: string;                     // DTMF keys with pause chars (e.g., "w1", "w2w1")
  spokenResponse?: string;           // What to say if action is "speak"
  expectedDelayMs?: number;          // Expected delay before this menu plays
  children?: PhoneTreeNode[];        // Sub-menu nodes
  categoryMatchSlugs?: string[];     // Which issue categories this path serves
  isHoldQueue?: boolean;             // True = reached the target hold queue
  adminNotes?: string;               // Notes for admin about this node
}

export interface IspPhoneTreeConfig {
  nodes: PhoneTreeNode[];                   // Root-level menu nodes
  holdInterruptionPatterns?: string[];      // Known hold interruption text to ignore
  navigationNotes?: string;                 // ISP-specific navigation notes for AI
  dtmfPauseChar?: "w" | "W";               // Preferred pause timing ("w"=0.5s default, "W"=1s)
}
```

Add the `ispPhoneTree` table to `src/db/schema.ts`:
- Import `jsonb`, `uniqueIndex`, `integer` (integer already imported)
- Import `IspPhoneTreeConfig` from `@/lib/phone-tree-types`
- Add the table AFTER the `issueCategory` table and BEFORE the `call` table:

```typescript
export const ispPhoneTree = pgTable("isp_phone_tree", {
  id: uuid("id").primaryKey().defaultRandom(),
  ispId: uuid("isp_id")
    .notNull()
    .references(() => isp.id),
  version: integer("version").notNull().default(1),
  tree: jsonb("tree").notNull().$type<IspPhoneTreeConfig>(),
  notes: text("notes"),
  updatedAt: timestamp("updated_at").notNull().defaultNow(),
  createdAt: timestamp("created_at").notNull().defaultNow(),
}, (table) => [
  uniqueIndex("isp_phone_tree_isp_idx").on(table.ispId),
]);
```
